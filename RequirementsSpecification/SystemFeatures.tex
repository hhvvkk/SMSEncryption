%All the system features, i.e. all the functional requirements for the application

\textbf{\\}


%%%%%%%%%%%%%%%
%%Messaging: The messaging requirements
%%%%%%%%%%%%%%%
\Large{\textbf{Messaging}}\\
\normalsize
\textbf{\\}


\textbf{Create message : FRQ1}\\%meaning functional requirement 1
\textbf{(Source: Bernard Wagner, Priority: High)}\\
\begin{itemize}
\item A message must be creatable in the application.
\item The message must be editable.
\end{itemize}

\textbf{\\}

\textbf{Encrypt message : FRQ2}\\%meaning functional requirement 2
\textbf{(Source: Bernard Wagner, Priority: High)}\\
\begin{itemize}
\item After that the message must be encrypted using a suitable encryption method.
\item The user must be able to select and copy the message to the clipboard in order to be sent using the method the user wants to.
\end{itemize}

\textbf{\\}

\textbf{AppSomething : FRQ3}\\%meaning functional requirement 3

\end{itemize}
